    <!-- 引入 GSAP 动画库 -->
    <script src="https://cdnjs.cloudflare.com/ajax/libs/gsap/3.12.2/gsap.min.js"></script>
    <script src="https://cdnjs.cloudflare.com/ajax/libs/gsap/3.12.2/ScrollTrigger.min.js"></script>

    <script>
        // 注册 ScrollTrigger 插件
        gsap.registerPlugin(ScrollTrigger);

        // 1. 自定义光标逻辑
        const cursorDot = document.querySelector('.cursor-dot');
        const cursorOutline = document.querySelector('.cursor-outline');
        
        window.addEventListener('mousemove', (e) => {
            const posX = e.clientX;
            const posY = e.clientY;

            // 点跟随（无延迟）
            cursorDot.style.left = `${posX}px`;
            cursorDot.style.top = `${posY}px`;

            // 圈跟随（GSAP实现平滑延迟）
            gsap.to(cursorOutline, {
                x: posX,
                y: posY,
                duration: 0.15,
                ease: "power2.out"
            });
        });

        // 悬停交互效果
        const hoverTriggers = document.querySelectorAll('.hover-trigger');
        hoverTriggers.forEach(trigger => {
            trigger.addEventListener('mouseenter', () => {
                const text = trigger.getAttribute('data-cursor-text');
                
                // 变大并改变颜色
                gsap.to(cursorOutline, {
                    width: 80,
                    height: 80,
                    backgroundColor: 'rgba(255, 255, 255, 0.2)',
                    borderColor: 'transparent',
                    duration: 0.3
                });
                
                if(text) {
                    // 如果有文字需求（这里简化处理，实际需添加内部文字DOM）
                    // cursorOutline.innerHTML = text; 
                }
            });

            trigger.addEventListener('mouseleave', () => {
                gsap.to(cursorOutline, {
                    width: 40,
                    height: 40,
                    backgroundColor: 'transparent',
                    borderColor: 'rgba(62, 60, 58, 0.5)',
                    duration: 0.3
                });
            });
        });

        // 2. 首页 Hero 动画 (入场)
        const tl = gsap.timeline();
        
        tl.from(".hero-text-area", {
            y: 50,
            opacity: 0,
            duration: 1.2,
            ease: "power3.out",
            stagger: 0.2
        })
        .from(".hero-img-area", {
            scale: 0.9,
            opacity: 0,
            duration: 1.5,
            ease: "expo.out"
        }, "-=1")
        .from(".floating-card", {
            y: 100,
            opacity: 0,
            rotation: 10,
            duration: 1.5,
            ease: "back.out(1.7)"
        }, "-=1");

        // 3. 滚动视差效果 (Parallax)
        gsap.utils.toArray('.img-container').forEach(container => {
            const img = container.querySelector('.img-parallax');
            
            gsap.to(img, {
                yPercent: 20, // 图片向下移动20%
                ease: "none",
                scrollTrigger: {
                    trigger: container,
                    start: "top bottom", // 容器顶部到达视口底部开始
                    end: "bottom top",   // 容器底部离开视口顶部结束
                    scrub: true          // 绑定滚动条进度
                }
            });
        });

        // 4. 文字渐显效果
        gsap.utils.toArray('.reveal-text').forEach(text => {
            gsap.to(text, {
                opacity: 1,
                duration: 1,
                scrollTrigger: {
                    trigger: text,
                    start: "top 80%",
                    end: "top 40%",
                    scrub: true
                }
            });
        });

        // 5. 画廊卡片错落浮现
        gsap.utils.toArray('.gallery-item').forEach((item, i) => {
            gsap.from(item, {
                y: 100,
                opacity: 0,
                duration: 1,
                ease: "power3.out",
                scrollTrigger: {
                    trigger: item,
                    start: "top 90%"
                }
            });
        });

    </script>
